int x = 100;

fn prueba() {
    int x = 5;          // ⚠️ Oculta variable global x
    if [x < 10] then
        int y = 2;
        x = y + x;
    fi

    y = 10;             // ❌ 'y' fuera de alcance
}

fn main() {
    x = x + 1;          // ✅ válido (usa la global)
    y = 2;              // ❌ 'y' no declarada
    prueba();
}
