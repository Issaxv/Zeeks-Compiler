fn main() {
    int a = 10;
    string b = "hola";
    float c = 3.5;

    a = b;              // ❌ Asignación incompatible string → int
    b = c;              // ❌ Asignación incompatible float → string
    c = a + b;          // ❌ Operación entre int y string
    a = (a + c) * b;    // ❌ Multiplicación entre número y string
}
